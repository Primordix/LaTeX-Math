% ========= interne Einstellungen =========
\usepackage{titlesec}
% PART
\titleformat{\part}[display]
{\centering\normalfont\huge\bfseries\color{graycolor}}
{\partname\ \thepart}
{20pt}
{\Huge\color{maincolor}}

% CHAPTER
\titleformat{\chapter}[display]
{\bfseries\Huge\color{maincolor}}
{\filleft\Large\color{graycolor}\chaptertitlename\ \thechapter}
{1ex}
{\titlerule[2pt]\vspace{1ex}\filleft}
[\vspace{1ex}\titlerule]

% SECTION
\titleformat{\section}
{\Large\bfseries\sffamily\color{maincolor}}
{\thesection}
{1em}
{}
[{\titlerule[0.8pt]}]

% SUBSECTION
\titleformat{\subsection}
{\large\bfseries\sffamily\color{accentcolor}}
{\thesubsection}
{1em}
{}

% SUBSUBSECTION
\titleformat{\subsubsection}
{\normalsize\bfseries\sffamily\color{black}}
{\thesubsubsection}
{1em}
{}

{}
{\titlerule*[0.7pc]{.}\contentspage}
